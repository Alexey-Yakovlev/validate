Input Validator: Examples

Input validator is a modern lightweight library without dependencies for data validation from single input tag.

Site Github

Required field

validate.prototype.required()

Code for this validation:

validate(input)
  .required('Try to type something in here')
  .onSuccess(function() {
    this.success();
  })
  .onError(function() {
    this.showErrors();
    this.error();
  });

Field should contain at least 8 characters

validate.prototype.min()

Code for this validation:

validate(input)
  .min(8, '%s characters required. Try typing "JavaScript"')
  .onSuccess(function() {
    this.success();
  })
  .onError(function() {
    this.showErrors();
    this.error();
  });

Field should contain less than 8 characters

validate.prototype.max()

Code for this validation:

validate(input)
  .max(8, 'You should not type more than %s characters. Try typing "banana"')
  .onSuccess(function() {
    this.success();
  })
  .onError(function() {
    this.showErrors();
    this.error();
  });

Field should contain '.js' substring

validate.prototype.contain()

Code for this validation:

validate(input)
  .contain('.js', 'This input should contain %s substring. Try "ember.js"')
  .onSuccess(function() {
    this.success();
  })
  .onError(function() {
    this.showErrors();
    this.error();
  });

Field should contain valid email address

validate.prototype.match()

Code for this validation:

validate(match.input)
  .match('email', 'This input should contain %s. Try email@example.com')
  .onSuccess(function() {
    this.success();
  })
  .onError(function() {
    this.showErrors();
    this.error();
});

All together

Code for this validation:

validate(allTogether.input)
  .min(8)
  .max(50)
  .match('email')
  .contain('gmail')
  .onSuccess(function() {
    this.success();
  })
  .onError(function() {
    this.showErrors();
    this.error();
});
